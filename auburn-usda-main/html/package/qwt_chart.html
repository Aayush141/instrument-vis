import time 

from qwt import QwtPlot, QwtPlotCurve, QwtPlotItem, QwtPlotMarker, QwtInterval
from PySide6.QtCore import Slot, Qt, QRect, QSize, Signal
from PySide6.QtGui import QPen, QColor, QMouseEvent, QPainter
from PySide6.QtWidgets import QRubberBand


from package.utils.data_classes import DataPoint, Annotation
from package.utils.utils import formatEpochTimeToClockTime
from package.utils.enums import AnnotationType
from package.range_marker import RangeMarker

class Chart(QwtPlot):
    """ Chart that contains data from EPG device """
    createAnnotationFromChart = Signal(int, int)

    def __init__(self, parent=None):
        QwtPlot.__init__(self, parent)
        # Set up canvas where chart will be drawn
        self.setTitle("This is an example")
        self.enableAxis(QwtPlot.yLeft, "Voltage (V)")
        self.enableAxis(QwtPlot.xBottom, "Time (s)")
        self.setAxisAutoScale(QwtPlot.yLeft, False)
        self.setAxisAutoScale(QwtPlot.xBottom, False)
        self.setAutoFillBackground(True)
        self.setStyleSheet("QwtPlotCanvas   {background-color: #FFFFFF}")
        self.curve = QwtPlotCurve()
        self.curve.attach(self)

        self.times = []
        self.voltages = []

        # Axes
        self.xRange = 30
        self.xMin = 0
        self.xMax = self.xRange
        self.setAxisScale(QwtPlot.xBottom, self.xMin, self.xMax)
        self.yMin = float("inf")
        self.yMax = float("-inf")

        self.samplesPerSecond = 100

        self.rubberBand = None


    def mousePressEvent(self, event: QMouseEvent):
        """ On mouse press, create resizable rectangle that represents
            the duration of a new annotation starting at clicked point

        :param event: QMouseEvent object with coordinates of mouse location
        """
        self.origin = event.pos()
        if not self.rubberBand:
            self.rubberBand = QRubberBand(QRubberBand.Shape.Rectangle, self)
        self.rubberBand.setGeometry(QRect(self.origin, QSize()))
        self.rubberBand.show()

    def mouseMoveEvent(self, event: QMouseEvent):
        """ On mouse move, update size of the resizable rectangle

            :param event: QMouseEvent object with coordinates of mouse location
        """
        self.rubberBand.setGeometry(QRect(self.origin, event.pos()).normalized())

    def mouseReleaseEvent(self, event: QMouseEvent):
        """ On mouse release, create the annotation according to the duration
            indicated by the resizable rectangle

            :param event: QMouseEvent object with coordinates of mouse location
        """
        self.rubberBand.hide()
        # Convert pixel coordinates to chart coordinates
        startX = self.invTransform(QwtPlot.xBottom, self.origin.x())
        endX = self.invTransform(QwtPlot.xBottom, event.pos().x())
        # Send chart coordinates to create annotation window
        self.createAnnotationFromChart.emit(startX, endX)

    def displayImportedData(self, data: list[DataPoint]):
        """Displays data from an imported EPG file on chart
        :param data: List of DataPoints
        """
        self.times = [point.time for point in data]
        self.voltages = [point.voltage for point in data]
        print("Plotting...")
        self.curve.setData(self.times, self.voltages)
        self.curve.attach(self)
        self.setAxisAutoScale(QwtPlot.xBottom, True)
        self.setAxisAutoScale(QwtPlot.yLeft, True)

    @Slot(DataPoint)
    def addDataPoints(self, points: list[DataPoint]):
        """Adds a new data point to a pre-existing chart and updates accordingly
        :param points: list of DataPoints retrieved from live serial data
        """

        yChanged = False 
        xChanged = False 

        for point in points:
            self.times.append(point.time)
            self.voltages.append(point.voltage)
            # Shift x-axis if points go off screen
            if point.time > self.xMax:
                self.xMin += self.xRange
                self.xMax += self.xRange
                xChanged = True 
            # Adjust y-axis
            if self.yMin > point.voltage:
                self.yMin = point.voltage
                yChanged = True 
            if self.yMax < point.voltage:
                self.yMax = point.voltage
                yChanged = True

        self.curve.setData(self.times, self.voltages)
        if xChanged:
            self.setAxisScale(QwtPlot.xBottom, self.xMin, self.xMax)
        if yChanged:
            self.setAxisScale(QwtPlot.yLeft, self.yMin, self.yMax + 40)
        self.replot()

    @Slot()
    def changeLineColorToRed(self):
        # Start new curve to change line color
        self.resetCurve(lastTime = self.times[-1], lastVoltage = self.voltages[-1])
        #self.createVerticalLineMarker(QColor("orange"), self.times[-1])
        self.curve.setPen(QPen(Qt.GlobalColor.gray))

    def savePausedRecording(self, savePaused:bool, buffer:list[DataPoint], timePausedFor):
        currentTime = time.time()
        firstPoint = buffer[0]
        if savePaused:
            self.curve.setPen(QPen(Qt.GlobalColor.black))
        else:
            # Clear paused portion of graph
            self.curve.detach() # is this needed
            self.replot()
            # self.createVerticalLineMarker(Qt.GlobalColor.red, firstPoint.time)
            self.resetCurve(firstPoint.time, firstPoint.voltage)
            
        self.addDataPoints(buffer)

    
    def resetCurve(self, lastTime: float, lastVoltage: float):
        """Clears current curve from graph and creates a new one
        """
        # Take last point of previous curve to make old curve and 
        # new curve "connected"
        self.times = [lastTime] 
        self.voltages = [lastVoltage] 
        self.curve = QwtPlotCurve()
        self.curve.setPen(QPen(Qt.GlobalColor.black))
        self.curve.attach(self)
    
    def clearChart(self):
        """ Erase all curves and markers from the chart
        """
        self.detachItems(QwtPlotItem.Rtti_PlotCurve)
        self.detachItems(QwtPlotItem.Rtti_PlotMarker)

        self.curve = QwtPlotCurve()
        self.curve.attach(self)

        self.times = []
        self.voltages = []
        self.curve.setData(self.times, self.voltages)

        self.xRange = 30
        self.xMin = 0
        self.xMax = self.xRange
        self.setAxisScale(QwtPlot.xBottom, self.xMin, self.xMax)

        self.yMin = float("inf")
        self.yMax = float("-inf")
